        function updateDeckView() {

            deck.innerHTML = ''; // Clear previous slides

            renderSlide(currentSlideIndex); // Render only the current one

            const currentSlideEl = deck.querySelector('.slide-container');

            if (currentSlideEl) {

                currentSlideEl.classList.add('active');

            }


            slideCounter.textContent = `Page ${currentSlideIndex + 1} of ${slidesData.length}`;

            prevBtn.disabled = currentSlideIndex === 0;

            nextBtn.disabled = currentSlideIndex === slidesData.length - 1;

        }


        function init() {

            updateDeckView();

        }


        prevBtn.addEventListener('click', () => {

            if (currentSlideIndex > 0) {

                currentSlideIndex--;

                updateDeckView();

            }

        });


        nextBtn.addEventListener('click', () => {

            if (currentSlideIndex < slidesData.length - 1) {

                currentSlideIndex++;

                updateDeckView();

            }

        });


        init();

    });
